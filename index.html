function drawGauge(percent){ const canvas=document.getElementById("gauge") const ctx=canvas.getContext("2d") ctx.clearRect(0,0,200,120) ctx.lineWidth=20 ctx.beginPath() ctx.strokeStyle="#eee" ctx.arc(100,100,70,Math.PI,0) ctx.stroke() ctx.beginPath() let end=Math.PI+(Math.PI*percent/100) ctx.strokeStyle="#4facfe" ctx.arc(100,100,70,Math.PI,end) ctx.stroke() ctx.font="20px Arial" ctx.fillStyle="#333" ctx.textAlign="center" ctx.fillText(percent+"%",100,95) }
